Simple Cryptography Project

Python Tkinter GUI

A simple cryptography project that converts normal text to encoded messages

# Cryptography code sample
import tkinter as tk
from tkinter import messagebox

def encrypt(text, shift):
    result = ""
    for char in text:
        if char.isalpha():
            ascii_val = ord(char)
            shifted = (ascii_val - 65 + shift) % 26 + 65 if char.isupper() else (ascii_val - 97 + shift) % 26 + 97
            result += chr(shifted)
        else:
            result += char
    return result
